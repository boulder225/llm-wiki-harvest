Create src/claude_dump/exporter.py with a single public function:

```python
"""Export pipeline: fetch conversations, render to Markdown, write to disk."""

from __future__ import annotations

from pathlib import Path
from typing import TYPE_CHECKING

from rich.progress import Progress, SpinnerColumn, TextColumn, BarColumn, MofNCompleteColumn

from claude_dump.markdown import make_filename, render_conversation
from claude_dump.models import SessionExpiredError

if TYPE_CHECKING:
    from claude_dump.client import ClaudeAPIClient


def export_project(
    client: ClaudeAPIClient,
    project_uuid: str,
    project_name: str,
    output_dir: str | Path,
) -> tuple[int, int]:
    """Export all conversations in a project to Markdown files.

    Args:
        client: Authenticated API client with org_id set.
        project_uuid: UUID of the project to export.
        project_name: Human-readable project name (for console output).
        output_dir: Root output directory. Conversations go into {output_dir}/conversations/.

    Returns:
        Tuple of (exported_count, failed_count).
    """
    output_path = Path(output_dir)
    conv_dir = output_path / "conversations"
    conv_dir.mkdir(parents=True, exist_ok=True)

    # Step 1: List all conversations (per D-02, pagination handled by client)
    conversations = client.list_conversations(project_uuid)

    if not conversations:
        return (0, 0)

    exported = 0
    failed = 0

    # Step 2: Fetch each conversation sequentially (per D-03), render, and write
    with Progress(
        SpinnerColumn(),
        TextColumn("[progress.description]{task.description}"),
        BarColumn(),
        MofNCompleteColumn(),
        transient=False,
    ) as progress:
        task_id = progress.add_task(
            f"Exporting {project_name}",
            total=len(conversations),
        )

        for conv_meta in conversations:
            # Update progress description (per D-15)
            progress.update(
                task_id,
                description=f"Exporting: {conv_meta.name or 'Untitled'}",
            )

            try:
                # Fetch full conversation with messages (per D-01)
                full_conv = client.get_conversation(conv_meta.uuid)

                # Render to Markdown
                markdown = render_conversation(full_conv)

                # Generate filename (per D-13, D-14)
                filename = make_filename(full_conv)

                # Write immediately to disk (per D-16, write-as-you-go)
                filepath = conv_dir / filename
                filepath.write_text(markdown, encoding="utf-8")

                exported += 1

            except SessionExpiredError:
                # Per D-17: halt immediately, already-exported files remain intact
                progress.stop()
                raise

            except Exception:
                # Non-fatal: skip this conversation, continue with rest
                failed += 1

            finally:
                progress.advance(task_id)

    return (exported, failed)
```

Key design decisions in this implementation:
- **Sequential fetching** (per D-03): no concurrency, one conversation at a time
- **Write-as-you-go** (per D-16): each file written immediately after rendering via `filepath.write_text()`
- **conversations/ subfolder** (per D-12): `conv_dir = output_path / "conversations"`
- **Session expiry halts** (per D-17): SessionExpiredError re-raised after stopping progress, letting CLI handle it. Already-written files are on disk.
- **Non-fatal errors logged** but don't stop export: increment failed counter, continue
- **Rich progress bar** (per D-15): shows spinner, description with conversation name, bar, and N/M counter
- Returns `(exported, failed)` tuple so CLI can report summary
