```python
class ContentBlock(BaseModel):
    model_config = ConfigDict(extra="ignore")
    type: str
    text: str = ""
    thinking: str = ""
    name: str = ""
    input: dict = Field(default_factory=dict)
    content: list[dict] = Field(default_factory=list)

class Attachment(BaseModel):
    model_config = ConfigDict(extra="ignore")
    file_name: str = ""
    file_type: str = ""
    extracted_content: str = ""

class FileRef(BaseModel):
    model_config = ConfigDict(extra="ignore")
    file_uuid: str = ""
    file_name: str = ""
    file_kind: str = ""

class ChatMessage(BaseModel):
    model_config = ConfigDict(extra="ignore")
    uuid: str = ""
    sender: str
    created_at: str = ""
    content: list[ContentBlock] = Field(default_factory=list)
    attachments: list[Attachment] = Field(default_factory=list)
    files_v2: list[FileRef] = Field(default_factory=list)

class Conversation(BaseModel):
    model_config = ConfigDict(extra="ignore")
    uuid: str
    name: str = ""
    model: str = ""
    created_at: str = ""
    updated_at: str = ""
    summary: str = ""
    chat_messages: list[ChatMessage] = Field(default_factory=list)
```

     ```
     > {attachment.extracted_content (first 500 chars)}
     ```

   - Return the complete message string

     ```
     uuid: {conv.uuid}
     ---
     ```

   - Add blank line, then `# {conv.name}` as the document heading
   - Add blank line
   - For each message in conv.chat_messages, call render_message and join with `\n\n---\n\n` (horizontal rule separator between messages)
   - Return the complete document string
